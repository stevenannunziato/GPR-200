#version 330

// varyings
in vec4 finalColor;
in vec2 texCoord;

layout (location = 0) out vec4 rtFragColor;

// uniforms
uniform sampler2D uParticleTex;

void main() {
	
	// sample particle texture
	vec4 texColor = texture(uParticleTex, texCoord);
	
	// finalize fragment color
	rtFragColor = texColor * finalColor;
	
}
