#version 300 es

layout (location = 0) in vec4 aPosition;
layout (location = 1) in vec3 aNormal;

uniform mat4 uModelMat, uViewMat, uProjMat;
uniform float uTime;

out vec4 vNormal;

void main()
{
	// apply some vertex deformation
	vec4 posWorld = uModelMat * aPosition;
	posWorld.x += sin(uTime * 2.0) * posWorld.y * 0.08;
	posWorld.z += cos(uTime * 2.0) * posWorld.y * 0.08;
	posWorld.y += sin(uTime * 1.0) * posWorld.y * 0.1;
	// convert to clip-space
	gl_Position = uProjMat * uViewMat * posWorld;
	
	vNormal = vec4(aNormal, 1.0);
}
