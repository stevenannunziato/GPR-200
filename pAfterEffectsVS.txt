#version 450

// attributes
layout (location = 0) in vec4 aPosition;
//layout (location = 2) in vec2 aTexcoord; does not exist for a ScreenQuadNDC

uniform mat4 uModelMatrix, uViewMatrix, uProjectionMatrix;

out vec4 vPosClip;
out vec2 vTexCoord;

void main()
{
	gl_Position = aPosition;
	// pass clip pos
	vec4 clipPos = uProjectionMatrix * uViewMatrix * uModelMatrix * aPosition;
	vPosClip = clipPos;
	vTexCoord = aPosition.xy;
}
