// BUFFER B: Texture pattern

const float numStripes = 10.0;

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    //fragColor = checkerboard(fragCoord, iResolution.xy);
    
    vec2 invRes = 1.0 / iResolution.xy;
    vec2 uv = fragCoord * invRes;
    float t = fragCoord.y * invRes.y * 6.28;
    uv.x += sin(t + iTime) * 1.0;
    
    // draw stripes
    fragColor = vec4(mod(floor(numStripes * (uv.x)), 2.0));
    fragColor.a = 1.0;
    
}
