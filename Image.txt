{

    sCoord uv = fragCoord / iChannelResolution[0].xy;

    //fragColor = sceneColor;
    //fragColor = bloomColor;
    
    // add
    //fragColor = sceneColor + bloomColor;
    
    // mix
    //fragColor = mix(sceneColor, bloomColor, 0.5);
    
    // screen
    fragColor = vec4(1.0) - (vec4(1.0) - sceneColor) * (vec4(1.0) - bloomColor);

}
